9/// Generate project template to given target
10pub fn gen(target: &Path) {
11    let target = target.to_path_buf();
12
13    // Skip if target already exists
14    if target.exists() {
15        info!("target already exists");
16        return;
17    }
18
19    fs::create_dir_all(&target).unwrap_or_else(|_| panic!("failed to generate {:?}", &target));
20    info!("creating path: {:?}", target);
21
22    let config = Config::default();
23    let config_str = serde_yaml::to_string(&config).unwrap();
24
25    // Write config_str to target/config.yml
26    let config_path = target.join("config.yml");
27    fs::write(config_path.clone(), config_str)
28        .unwrap_or_else(|_| panic!("failed to write {:?}", config_path));
29    info!("Generated: {:?}", config_path);
30}

32/// Generating password with given length
33pub fn gen_password(
34    length: u8,
35    no_special: bool,
36    username: Option<String>,
37    password: Option<String>,
38) {
39    // If not password is given, generate random password
40    let password = match password {
41        Some(p) => p,
42        None => {
43            let chars: &[u8] = if no_special {
44                b"ABCDEFGHIJKLMNOPQRSTUVWXYZ\
45                  abcdefghijklmnopqrstuvwxyz\
46                  0123456789"
47            } else {
48                b"ABCDEFGHIJKLMNOPQRSTUVWXYZ\
49                  abcdefghijklmnopqrstuvwxyz\
50                  0123456789)(*&^%#!~"
51            };
52            let mut rng = rand::thread_rng();
53            let password: String = (0..length)
54                .map(|_| {
55                    let idx = rng.gen_range(0..chars.len());
56                    chars[idx] as char
57                })
58                .collect();
59
60            password
61        }
62    };
63
64    println!("Generated password: {}", Green.paint(password.clone()));
65
66    if let Some(username) = username {
67        let password_hash = gen_md5_password(&password, &username);
68        println!(
69            "Generated MD5 (user: {}): {}",
70            username,
71            Green.paint(password_hash)
72        );
73        println!("\nHint: https://docs.aws.amazon.com/redshift/latest/dg/r_CREATE_USER.html");
74    } else {
75        println!("\nHint: Please provide --username to generate MD5");
76    }
77}

79/// Generate md5 password hash from username and password
80/// 1. Concatenate the password and username
81/// 2. Hash the concatenated string
82/// 3. Concatenate 'md5' in front of the MD5 hash string
83/// https://docs.aws.amazon.com/redshift/latest/dg/r_CREATE_USER.html
84fn gen_md5_password(password: &str, username: &str) -> String {
85    format!(
86        "md5{:x}",
87        compute(format!("{}{}", password, username).as_bytes())
88    )
89}
